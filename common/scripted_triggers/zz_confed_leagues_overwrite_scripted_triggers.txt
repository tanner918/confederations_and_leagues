# Up-to-date as of CK3 1.18.2

valid_confederation_member_trigger = {
	is_playable_character = yes
	is_tributary = no
	is_landed = yes
	is_independent_ruler = yes
	is_alive = yes
	NOR = {
		has_trait = conqueror
		has_trait = greatest_of_khans
		is_at_war_with = $CHARACTER$
		is_at_war_with = $CHARACTER$.top_liege
		custom_tooltip = {
			text = no_raiding_hostility_tt
			$CHARACTER$ = {
				is_causing_raid_hostility_towards = prev
			}
		}
	}
	CL_has_appropriate_title_tier = yes # C&L: Changed this from allowing just duchies and below to instead change depending on what game rule the player has set
	CL_has_appropriate_realm_laws = yes # C&L: Base game just checks if you are nomadic and have the relevant realm laws, despite them also putting restrictions on tribal realm laws. This checks against all relevant realm laws for each government
	custom_tooltip = { # C&L: Base game just checks if you are Nomadic/Tribal. Changed it so that it uses the custom scripted trigger in case more governments are allowed in later
		CL_uses_confederations = yes
		text = CL_has_valid_confederation_government_tt # C&L: Base game uses is_nomadic_or_tribal_tt. Changed it to better show that more governments can join
	}
	is_confederation_member = no
}
